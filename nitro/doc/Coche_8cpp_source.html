391 bool Coche::isEncimaNodo(string nodo, OgreBulletDynamics::DynamicsWorld* world) {
392  bool meta = false;
393  Vector3 posicion = getCarChassisPtr()->getCenterOfMassPosition ();
394  btVector3 posIni(posicion.x, posicion.y, posicion.z);
395  btVector3 posFin(posicion.x, posicion.y - 1, posicion.z);
396 
397  // Start and End are vectors
398  btCollisionWorld::ClosestRayResultCallback RayCallback(posIni, posFin);
399 
400  // Perform raycast
401  world->getBulletDynamicsWorld ()->rayTest(posIni, posFin, RayCallback);
402 
403  if(RayCallback.hasHit()) {
404  btCollisionObject* obA = (btCollisionObject*)(RayCallback.m_collisionObject);
405  OgreBulletCollisions::Object *obOB_A = world->findObject(obA);
406 
407  meta = obOB_A->getRootNode()->getName() == nodo;
408  }
409 
410  return meta;
411 }
